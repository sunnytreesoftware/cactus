JavaScript Code (sample)

script.js

function calculateRatio() {

   // Wikipedia Formula
   var d = ((a + b) / c) + ":";
   $('span#form2').text(d+e);

   // Bankrate Formula
   var f = ((a - b) / c) + ":";
   $('span#form1').text(f+e);
}
